28  const std::string& metaFull = "__meta_" + name + "_" + type;
29  // make global
30  lua_setfield(s, LUA_REGISTRYINDEX, metaFull.c_str());
31  // put back onto stack
32  lua_getfield(s, LUA_REGISTRYINDEX, metaFull.c_str());
33

36  // this 'metatable' must not be in the global registry.
