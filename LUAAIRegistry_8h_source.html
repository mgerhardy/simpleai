170  static void setupMetatable(lua_State* s, const std::string& type, const luaL_Reg *funcs, const std::string& name) {
171  const std::string& metaFull = "__meta_" + name + "_" + type;
172  // make global
173  lua_setfield(s, LUA_REGISTRYINDEX, metaFull.c_str());
174  // put back onto stack
175  lua_getfield(s, LUA_REGISTRYINDEX, metaFull.c_str());
176 
177  // setup meta table - create a new one manually, otherwise we aren't
178  // able to override the particular function on a per instance base. Also
179  // this 'metatable' must not be in the global registry.
180  lua_createtable(s, 0, 2);
181  lua_pushvalue(s, -1);
182  lua_setfield(s, -2, "__index");
183  lua_pushstring(s, name.c_str());
184  lua_setfield(s, -2, "__name");
185  lua_pushstring(s, type.c_str());
186  lua_setfield(s, -2, "type");
187  luaL_setfuncs(s, funcs, 0);
188  lua_setmetatable(s, -2);
189  }
